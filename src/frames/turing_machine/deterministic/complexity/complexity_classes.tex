\begin{frame}{Classes de complexité importantes}
  \small
  \begin{tikzpicture}
    \draw (0,9.5) node{\scalebox{.85}{\begin{minipage}{.6\textwidth}
          \begin{block}{Décision en temps (MTD)}
            \begin{description}
            \item[Polynomial :]
              $\alert{\displaystyle \textsc{p} = \bigcup_{k\in \mathbb{N}} \textsc{time}(n^k)}$
            \item[Exponentiel :]
              $\alert{\displaystyle \textsc{exptime} = \bigcup_{k\in \mathbb{N}} \textsc{time}(2^{n^k})}$
            \end{description}
          \end{block}
    \end{minipage}}};

    \draw (.5\textwidth,9.5) node{\scalebox{.85}{\begin{minipage}{.6\textwidth}
          \begin{block}{Reconnaissance en temps (MTND)}
            \begin{description}
            \item[Polynomial :]
              $\alert{\displaystyle \textsc{np} = \bigcup_{k\in \mathbb{N}} \textsc{ntime}(n^k)}$
            \item[Exponentiel :]
              $\alert{\displaystyle \textsc{nexptime} = \bigcup_{k\in \mathbb{N}} \textsc{ntime}(2^{n^k})}$
            \end{description}
          \end{block}
    \end{minipage}}};
    

    \draw (0,7) node{\scalebox{.85}{\begin{minipage}{.6\textwidth}
          \begin{block}{Décision en espace (MTD)}
            \begin{description}
            \item[Polynomial :]
              $\alert{\displaystyle \textsc{pspace} = \bigcup_{k\in \mathbb{N}} \textsc{dspace}(n^k)}$
            \item[Exponentiel :]
              $\alert{\displaystyle \textsc{expspace} = \bigcup_{k\in \mathbb{N}} \textsc{dspace}(2^{n^k})}$
            \end{description}
          \end{block}
    \end{minipage}}};

    \draw (.5\textwidth,7) node{\scalebox{.85}{\begin{minipage}{.6\textwidth}
          \begin{block}{Reconnaissance en espace (MTND)}
            \begin{description}
            \item[Polynomial :]
              $\alert{\displaystyle \textsc{npspace} = \bigcup_{k\in \mathbb{N}} \textsc{nspace}(n^k)}$
            \item[Exponentiel :]
              $\alert{\displaystyle \textsc{nexpspace} = \bigcup_{k\in \mathbb{N}} \textsc{nspace}(2^{n^k})}$
            \end{description}
          \end{block}
    \end{minipage}}};

    \uncover<2>{
    \draw (2,4.5) node{\scalebox{.85}{\begin{minipage}{\textwidth}
          \begin{block}{Quelques résultats connus}
            \begin{itemize}
            \item $\textsc{p} \subseteq \textsc{np} \subseteq \textsc{exptime}  \subseteq \textsc{nexptime} \subseteq \textsc{expspace}$
            \item $\textsc{p} \neq \textsc{exptime}$
            \item $\textsc{np} \subseteq \textsc{pspace} = \textsc{npspace} \subsetneq \textsc{expspace} = \textsc{nexpspace}$
            \end{itemize}
          \end{block}
    \end{minipage}}};
    }
    
  \end{tikzpicture}

\end{frame}
