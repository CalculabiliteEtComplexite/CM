\begin{frame}{Générateur d'un langage}
  
  \onBlock[top=-4mm]{Définition -- Génération d'un langage}{
    Soit $M=\langle \Sigma, \Gamma, \blank, Q, q_0, F, \rightarrow \rangle$ une MTND.
    \begin{itemize}
    \item Le \structure{langage généré par $M$} est l'ensemble des mots écrits sur le ruban quand $M$ passe par un état accepteur à partir du mot vide:
      $$\alert{\mathcal{L}_G(M) = \left\{G\cdot D \in \Sigma^\star \;\middle\mid\; \exists q\in F,~ C_{\mathit{init}}(\varepsilon) \leadsto_M^\star \langle G, q, D \rangle \right\}}$$
    \item $M$ est appelée un \structure{générateur} du langage $\mathcal{L}_G(M)$
    \item \structure{Rappel :} Si $M$ est déterministe, alors $\mathcal{L}_G(M)$ est énuméré par $M$
    \end{itemize}
  }
  
  \onExampleBlock[y=-15mm]{Exemple : générateur du langage $\mathcal{L}(ba^\star b)$}{}
 
  \on[y=-25mm, x=-27mm, anchor=mid]{

    \end{tikzpicture}
  }
