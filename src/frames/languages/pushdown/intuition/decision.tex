\begin{frame}{Décision de l'appartenance à un langage algébrique}

  \begin{block}{Problème de décision}
    Soit $L$ un langage algébrique
    
    \Probleme{\structure{Decision$_L$}}{
      Un mot \structure{$u \in \Sigma^\star$}
    }{
      Est-ce que \structure{$u \in L$} ?
    }
  \end{block}

  \begin{alertblock}{Questions}
    \begin{itemize}
    \item Quelles structures algorithmiques permettent de décider \structure{\textsc{Decision}$_L$} ?
    \item À quel coût ?
    \end{itemize}
  \end{alertblock}
  
  \begin{block}{Automate à pile}
    \begin{itemize}
    \item Modélise un algorithme \structure{récursif} sans structure de donnée complexe
    \item La reconnaissance se fait par \structure{simulation non-déterministe}
    \end{itemize}
  \end{block}

\end{frame}
