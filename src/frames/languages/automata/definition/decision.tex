\begin{frame}{Décision de l'appartenance à un langage rationnel}

  \begin{block}{Problème de décision}
    Soit $L$ un langage rationnel
    
    \Probleme{\structure{Decision$_L$}}{
      Un mot \structure{$u \in \Sigma^\star$}
    }{
      Est-ce que \structure{$u \in L$} ?
    }
  \end{block}

  \begin{alertblock}{Questions}
    \begin{itemize}
    \item Quelles structures algorithmiques permettent de décider \structure{\textsc{Decision}$_L$} ?
    \item À quel coût ?
    \end{itemize}
  \end{alertblock}

  \begin{block}{Automate fini}
    \begin{itemize}
    \item Modélise un algorithme \structure{sans structure de donnée complexe}
    \item La reconnaissance se fait par \structure{simulation déterministe}
    \end{itemize}
  \end{block}

\end{frame}
