\begin{frame}{Conséquences : tout formalisme a des limites}
  \small
  Pour tous alphabets $\Sigma$ et $\Gamma$, il n'existe pas de langage $L \subseteq \Sigma^\star$ équipotent à $\mathscr{P}(\Gamma^\star)$.
  
  \begin{itemize}
  \item Soient $L$ un langage sur $\Sigma$ et $\mathcal{S} : L \rightarrow \mathscr{P}(\Gamma^\star)$ une fonction de sémantique.
  \item Alors il existe un langage $L'$ tel que $\forall u\in L, L' \neq \mathcal{S}(u)$.
  \end{itemize}

  \begin{exampleblock}{Exemples}
    \begin{itemize}
    \item Si $L$ est le langage des \structure{expressions rationnelles} (flex)
      \begin{itemize}
      \item Il existe des langages qui ne sont \alert{pas rationnels}
      \end{itemize}
    \item Si $L$ est le langage des \structure{grammaires algébriques} (bison)
      \begin{itemize}
      \item Il existe des langages qui ne sont \alert{pas algébriques}
      \end{itemize}
    \item Si $L$ est le langage des \structure{machines de Turing} (langage de programmation)
      \begin{itemize}
      \item Il existe des langages qui ne sont \alert{pas récursivement énumérables}
      \end{itemize}
    \item Si $L$ est le langage des \structure{machines de Turing qui terminent}
      \begin{itemize}
      \item Il existe des langages qui ne sont \alert{pas décidables}
      \end{itemize}
    \item Si $L$ est le langage des \structure{mathématiques}
      \begin{itemize}
      \item Il existe des langages qui ne sont \alert{pas définissables}
      \end{itemize}
    \end{itemize}
  \end{exampleblock}
\end{frame}
